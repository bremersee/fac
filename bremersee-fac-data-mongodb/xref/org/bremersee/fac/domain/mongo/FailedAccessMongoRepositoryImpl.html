17  package org.bremersee.fac.domain.mongo;
18  
19  import java.util.Date;
20  import java.util.List;
21  
22  import org.apache.commons.lang3.Validate;
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.data.mongodb.core.MongoOperations;
25  import org.springframework.data.mongodb.core.query.Criteria;
26  import org.springframework.data.mongodb.core.query.Query;
27  
28  /**
29   * @author Christian Bremer
30   *
31   */
32  public class FailedAccessMongoRepositoryImpl implements FailedAccessMongoRepositoryCustom {
33      
34      @Autowired
35      protected MongoOperations mongoOperations;
36  
37      public void setMongoOperations(MongoOperations mongoOperations) {
38          this.mongoOperations = mongoOperations;
39      }
40      
41      @Override
42      public long count(String searchValue) {
43          
44          Validate.notBlank(searchValue, "searchValue must not be null or blank");
45          Query query = new Query();
46          // MongoRegexCreator is available at 1.9.x
47  //        query.addCriteria(
48  //                Criteria.where("resourceId").regex(MongoRegexCreator.INSTANCE.toRegularExpression(searchValue, Part.Type.LIKE))
49  //                .orOperator(Criteria.where("remoteHost").regex(MongoRegexCreator.INSTANCE.toRegularExpression(searchValue, Part.Type.LIKE))));
50          query.addCriteria(
51                  Criteria.where("resourceId").regex(toLikeRegex(searchValue))
52                  .orOperator(Criteria.where("remoteHost").regex(toLikeRegex(searchValue))));
53          return mongoOperations.count(query, FailedAccessMongoDoc.class);
54      }
55  
56      @Override
57      public List<FailedAccessMongoDoc> findObsolete(long removeFailedAccessEntriesAfterMillis) {
58  
59          long modificationDate = System.currentTimeMillis() - removeFailedAccessEntriesAfterMillis;
60          Query query = new Query();
61          query.addCriteria(Criteria.where("modificationDate").lt(new Date(modificationDate)));
62          return mongoOperations.find(query, FailedAccessMongoDoc.class);
63      }
64      
65      private String toLikeRegex(String source) {
66          return source.replaceAll("\\*", ".*");
67      }
68  
69  }
